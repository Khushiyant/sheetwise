Quick Start

# 1. Import
from sheetwise import SpreadsheetLLM
import pandas as pd

# 2. Load Data
df = pd.read_excel("quarterly_report.xlsx")
sllm = SpreadsheetLLM()

# 3. Magic Happens
encoded_text = sllm.encode_to_token_limit(df, max_tokens=4000)

print(encoded_text)
